1.Создаём файл для срипта mkdir -p ~/bin.
Далее через редактор нано вставляем српит в файл.
Скрипт: #!/usr/bin/env bash
# Скрипт приветствует пользователя, показывает дату и время запуска,
# количество переданных аргументов и сообщает об успешности выполнения вывода.

name="${1:-$USER}"

echo "Здравствуйте, ${name}. Скрипт запущен: $(date '+%Y-%m-%d %H:%M:%S'). Количество переданных аргументов: $#."

if [ $? -eq 0 ]; then
    echo "Завершено успешно"
else
    echo "Завершено с ошибкой"
fi
